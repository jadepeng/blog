支持常用的语言，下面试下go语言

package main

import (
    "fmt"
    "math"
    "runtime"
)

func sqrt(x float64) string {
    if x < 0 {
        return sqrt(-x) + "i"
    }
    return fmt.Sprint(math.Sqrt(x))
}
